#### js中for-in语句

```html
<!DOCTYPE html>
<html>
	<head>
		<meta charset="UTF-8">
		<title>js之for-in语句</title>
		<script>
			/**
			 * for-in语句，类似于java中的增强for循环语句
			 * java:for(类型 变量：数组名称/对象名称（集合）){
			 * 	输出变量
			 * }
			 * js中for-in格式：
			 * for(var 变量名 in 数组对象/自定义对象){
			 * 	输出这个变量即可
			 * }
			 */
			//js中数组--->Array
			var arr=[10,20,30,40];
			//普通for循环方式
			for(var i=0;i<arr.length;i++){
				document.write(arr[i]+"&nbsp;&nbsp;")
			}
			document.write("<hr/>");
			//for-in
			for(var i in arr){
				document.write(arr[i]+"&nbsp;&nbsp;");
			}
			document.write("<hr/>");
			//自定义一个Person对象
			function Person(name,age){//this代表当前对象的地址在引用
				this.name=name;
				this.age=age;
			}
			//创建一个对象
			var p=new Person("罗志祥",18);
			//将Person对象的属性遍历出来：for-in:对象名[变量]
			for(var i in p){
				document.write(p[i]+"<br/>");
			}
			
			<!--
				在浏览器中输出:*形 5行
			* * * *
			* * * *
			...
			-->
			//with语句实现
			/**
			 * with语句格式
			 * with(文档对象：document){
			 * 	具体的逻辑；
			 * }
			 */
			with(document){
				for(var i=0;i<5;i++){
					for(var j=0;j<5;j++){
						write("*&nbsp;");
					}
					write("<br/>");
				}
			}
		</script>
	</head>
	<body>
		
	</body>
</html>
```
